15pub fn handle_send(
16    url: Option<String>,
17    method: String,
18    headers: Vec<(String, String)>,
19    body: Option<String>,
20    json: Option<String>,
21    data: Vec<(String, String)>,
22    timeout: Option<u64>,
23    follow_redirects: bool,
24    env: Option<String>,
25    verbose: bool,
26    from_collection: Option<String>,
27) {
28    let x = 10;
29    let storage = match Storage::new() {
30        Ok(s) => s,
31        Err(e) => {
32            print_error(&format!("Failed to initialize storage: {}", e));
33            return;
34        }
35    };
36
37    // Load request from collection if specified
38    let (final_url, final_method, final_headers, final_body, final_json) =
39        if let Some(ref collection_path) = from_collection {
40            let parts: Vec<&str> = collection_path.split('/').collect();
41            if parts.len() != 2 {
42                print_error("Invalid format. Use: collection_name/request_name");
43                return;
44            }
45
46            let collection_name = parts[0];
47            let request_name = parts[1];
48
49            let collections = storage.load_collections().unwrap_or_default();
50            let collection = collections.iter().find(|c| c.name == collection_name);
51
52            if collection.is_none() {
53                print_error(&format!("Collection '{}' not found", collection_name));
54                return;
55            }
56
57            let saved_request = collection
58                .unwrap()
59                .requests
60                .iter()
61                .find(|r| r.name == request_name);
62
63            if saved_request.is_none() {
64                print_error(&format!(
65                    "Request '{}' not found in collection '{}'",
66                    request_name, collection_name
67                ));
68                return;
69            }
70
71            let req = saved_request.unwrap();
72            let loaded_url = req.request.url.clone();
73            let loaded_method = req.request.method.clone();
74            let loaded_headers = req.request.headers.clone();
75            let loaded_body = req.request.body.clone();
76
77            // CLI args override collection values
78            let use_url = url.unwrap_or(loaded_url);
79            let use_headers = if headers.is_empty() {
80                loaded_headers
81            } else {
82                headers
83            };
84            let use_body = body.or(loaded_body.clone());
85            let use_json = json.or(loaded_body);
86
87            (use_url, loaded_method, use_headers, use_body, use_json)
88        } else {
89            if url.is_none() {
90                print_error("URL is required when not using --from-collection");
91                return;
92            }
93            (url.unwrap(), method, headers, body, json)
94        };
95
96    let env_set = storage.load_environment_set().unwrap_or_default();
97    let resolved_url = if let Some(ref env_name) = env {
98        if let Some(environment) = env_set.get(env_name) {
99            environment.resolve_template(&final_url)
100        } else {
101            print_error(&format!("Environment '{}' not found", env_name));
102            return;
103        }
104    } else {
105        final_url.clone()
106    };
107
108    let http_method = match final_method.to_uppercase().as_str() {
109        "GET" => HttpMethod::Get,
110        "POST" => HttpMethod::Post,
111        "PUT" => HttpMethod::Put,
112        "PATCH" => HttpMethod::Patch,
113        "DELETE" => HttpMethod::Delete,
114        "HEAD" => HttpMethod::Head,
115        "OPTIONS" => HttpMethod::Options,
116        _ => {
117            print_error(&format!("Invalid HTTP method: {}", final_method));
118            return;
119        }
120    };
121
122    let mut request = HttpRequest::new(http_method, &resolved_url);
123
124    for (key, value) in final_headers {
125        let resolved_key = if let Some(ref env_name) = env {
126            if let Some(environment) = env_set.get(env_name) {
127                environment.resolve_template(&key)
128            } else {
129                key.clone()
130            }
131        } else {
132            key.clone()
133        };
134
135        let resolved_value = if let Some(ref env_name) = env {
136            if let Some(environment) = env_set.get(env_name) {
137                environment.resolve_template(&value)
138            } else {
139                value.clone()
140            }
141        } else {
142            value.clone()
143        };
144
145        request = request.with_header(&resolved_key, &resolved_value);
146    }
147
148    if let Some(json_body) = final_json {
149        let resolved_body = if let Some(ref env_name) = env {
150            if let Some(environment) = env_set.get(env_name) {
151                environment.resolve_template(&json_body)
152            } else {
153                json_body.clone()
154            }
155        } else {
156            json_body.clone()
157        };
158        request = request.with_json_body(&resolved_body);
159    } else if !data.is_empty() {
160        let form_data: HashMap<String, String> = data.into_iter().collect();
161        request = request.with_body(RequestBody::FormUrlEncoded(form_data));
162    } else if let Some(text_body) = final_body {
163        let resolved_body = if let Some(ref env_name) = env {
164            if let Some(environment) = env_set.get(env_name) {
165                environment.resolve_template(&text_body)
166            } else {
167                text_body.clone()
168            }
169        } else {
170            text_body.clone()
171        };
172        request = request.with_text_body(&resolved_body);
173    }
174
175    if let Some(t) = timeout {
176        request = request.with_timeout(t);
177    }
178
179    request = request.follow_redirects(follow_redirects);
180
181    let config = HttpClientConfig {
182        timeout_seconds: timeout.unwrap_or(30),
183        follow_redirects,
184        ..Default::default()
185    };
186
187    let client = HttpClient::with_config(config);
188
189    let start = Instant::now();
190    match client.send(&request) {
191        Ok(response) => {
192            let duration_ms = start.elapsed().as_millis() as u64;
193
194            let history_request = HistoryRequest {
195                method: request.method.as_str().to_string(),
196                url: request.url.clone(),
197                headers: request
198                    .headers
199                    .iter()
200                    .map(|h| (h.name.clone(), h.value.clone()))
201                    .collect(),
202                body: match &request.body {
203                    RequestBody::None => None,
204                    RequestBody::Text(t) => Some(t.clone()),
205                    RequestBody::Json(j) => Some(j.clone()),
206                    _ => Some("<complex body>".to_string()),
207                },
208            };
209
210            let history_response = HistoryResponse::from_response(&response);
211            let entry = HistoryEntry::new(history_request, history_response, duration_ms);
212
213            if let Err(e) = storage.save_history_entry(&entry) {
214                eprintln!("Warning: Failed to save to history: {}", e);
215            }
216
217            print_response(&response, verbose);
218        }
219        Err(e) => print_error(&format!("Request failed: {}", e)),
220    }
221}

223pub fn handle_collection(action: CollectionAction) {
224    let storage = match Storage::new() {
225        Ok(s) => s,
226        Err(e) => {
227            print_error(&format!("Failed to initialize storage: {}", e));
228            return;
229        }
230    };
231
232    match action {
233        CollectionAction::New { name, description } => {
234            let collection = Collection {
235                name: name.clone(),
236                description,
237                folders: Vec::new(),
238                requests: Vec::new(),
239            };
240
241            match storage.save_collection(&collection) {
242                Ok(_) => print_success(&format!("Collection '{}' created", name)),
243                Err(e) => print_error(&format!("Failed to create collection: {}", e)),
244            }
245        }
246
247        CollectionAction::List => match storage.list_collections() {
248            Ok(collections) => {
249                if collections.is_empty() {
250                    print_info("No collections found");
251                } else {
252                    println!("\n{}:", "Collections".bold().cyan());
253                    for name in collections {
254                        println!("  • {}", name);
255                    }
256                    println!();
257                }
258            }
259            Err(e) => print_error(&format!("Failed to list collections: {}", e)),
260        },
261
262        CollectionAction::Show { name } => match storage.load_collection(&name) {
263            Ok(collection) => {
264                println!("\n{}: {}", "Collection".bold().cyan(), collection.name);
265                if let Some(desc) = &collection.description {
266                    println!("{}: {}", "Description".bold(), desc);
267                }
268                println!("\n{}:", "Requests".bold().cyan());
269                if collection.requests.is_empty() {
270                    println!("  {}", "(no requests)".bright_black());
271                } else {
272                    for req in &collection.requests {
273                        println!("  • {} - {}", req.name, req.request.url);
274                    }
275                }
276                println!();
277            }
278            Err(e) => print_error(&format!("Failed to load collection: {}", e)),
279        },
280
281        CollectionAction::Add {
282            collection,
283            name,
284            url,
285            method,
286            header,
287            body,
288            description,
289        } => {
290            let mut coll = match storage.load_collection(&collection) {
291                Ok(c) => c,
292                Err(_) => {
293                    print_error(&format!("Collection '{}' not found", collection));
294                    return;
295                }
296            };
297
298            let http_method = match method.to_uppercase().as_str() {
299                "GET" => HttpMethod::Get,
300                "POST" => HttpMethod::Post,
301                "PUT" => HttpMethod::Put,
302                "PATCH" => HttpMethod::Patch,
303                "DELETE" => HttpMethod::Delete,
304                "HEAD" => HttpMethod::Head,
305                "OPTIONS" => HttpMethod::Options,
306                _ => {
307                    print_error(&format!("Invalid HTTP method: {}", method));
308                    return;
309                }
310            };
311
312            let mut request = HttpRequest::new(http_method, &url);
313            for (key, value) in header {
314                request = request.with_header(&key, &value);
315            }
316            if let Some(b) = body {
317                request = request.with_text_body(&b);
318            }
319
320            let saved_request = SavedRequest {
321                id: uuid::Uuid::new_v4().to_string(),
322                name: name.clone(),
323                description,
324                request: SerializableRequest::from_request(&request),
325            };
326
327            coll.requests.push(saved_request);
328
329            match storage.save_collection(&coll) {
330                Ok(_) => print_success(&format!(
331                    "Request '{}' added to collection '{}'",
332                    name, collection
333                )),
334                Err(e) => print_error(&format!("Failed to save collection: {}", e)),
335            }
336        }
337
338        CollectionAction::Delete { name } => match storage.delete_collection(&name) {
339            Ok(_) => print_success(&format!("Collection '{}' deleted", name)),
340            Err(e) => print_error(&format!("Failed to delete collection: {}", e)),
341        },
342
343        CollectionAction::Export { name, output } => match storage.load_collection(&name) {
344            Ok(collection) => {
345                let json = match serde_json::to_string_pretty(&collection) {
346                    Ok(j) => j,
347                    Err(e) => {
348                        print_error(&format!("Failed to serialize collection: {}", e));
349                        return;
350                    }
351                };
352                match std::fs::write(&output, json) {
353                    Ok(_) => print_success(&format!("Collection exported to '{}'", output)),
354                    Err(e) => print_error(&format!("Failed to write file: {}", e)),
355                }
356            }
357            Err(e) => print_error(&format!("Failed to load collection: {}", e)),
358        },
359
360        CollectionAction::Import { input } => {
361            let contents = match std::fs::read_to_string(&input) {
362                Ok(c) => c,
363                Err(e) => {
364                    print_error(&format!("Failed to read file: {}", e));
365                    return;
366                }
367            };
368
369            // Try auto-detection of format
370            let imported_collections = match auto_import(&contents) {
371                Ok(collections) => collections,
372                Err(e) => {
373                    print_error(&format!("Failed to import: {}", e));
374                    return;
375                }
376            };
377
378            // Convert imported collections to native format and save
379            let mut success_count = 0;
380            let mut error_count = 0;
381
382            for imported in imported_collections {
383                let collection = convert_imported_to_collection(imported);
384                match storage.save_collection(&collection) {
385                    Ok(_) => {
386                        print_success(&format!("Imported collection '{}'", collection.name));
387                        success_count += 1;
388                    }
389                    Err(e) => {
390                        print_error(&format!(
391                            "Failed to save collection '{}': {}",
392                            collection.name, e
393                        ));
394                        error_count += 1;
395                    }
396                }
397            }
398
399            if success_count > 0 {
400                println!(
401                    "\n{} collection(s) imported successfully",
402                    success_count.to_string().green().bold()
403                );
404            }
405            if error_count > 0 {
406                println!(
407                    "{} collection(s) failed to import",
408                    error_count.to_string().red().bold()
409                );
410            }
411        }
412    }
413}

415pub fn handle_env(action: EnvAction) {
416    let storage = match Storage::new() {
417        Ok(s) => s,
418        Err(e) => {
419            print_error(&format!("Failed to initialize storage: {}", e));
420            return;
421        }
422    };
423
424    let mut env_set = storage.load_environment_set().unwrap_or_default();
425
426    match action {
427        EnvAction::List => {
428            let envs: Vec<String> = env_set
429                .environments
430                .iter()
431                .map(|e| e.name.clone())
432                .collect();
433            if envs.is_empty() {
434                print_info("No environments found");
435            } else {
436                println!("\n{}:", "Environments".bold().cyan());
437                for name in envs {
438                    let marker = if env_set.get_active().map(|e| &e.name) == Some(&name) {
439                        "* ".green()
440                    } else {
441                        "  ".normal()
442                    };
443                    println!("{}• {}", marker, name);
444                }
445                println!();
446            }
447        }
448
449        EnvAction::Set { name, variables } => {
450            let vars: HashMap<String, String> = variables.into_iter().collect();
451            let environment = Environment {
452                name: name.clone(),
453                variables: vars,
454            };
455            env_set.add(environment);
456
457            match storage.save_environment_set(&env_set) {
458                Ok(_) => print_success(&format!("Environment '{}' saved", name)),
459                Err(e) => print_error(&format!("Failed to save environment: {}", e)),
460            }
461        }
462
463        EnvAction::Show { name } => {
464            if let Some(env) = env_set.get(&name) {
465                println!("\n{}: {}", "Environment".bold().cyan(), env.name);
466                println!("\n{}:", "Variables".bold().cyan());
467                if env.variables.is_empty() {
468                    println!("  {}", "(no variables)".bright_black());
469                } else {
470                    for (key, value) in &env.variables {
471                        println!("  {} = {}", key.bright_white(), value);
472                    }
473                }
474                println!();
475            } else {
476                print_error(&format!("Environment '{}' not found", name));
477            }
478        }
479
480        EnvAction::Delete { name } => {
481            env_set.remove(&name);
482            match storage.save_environment_set(&env_set) {
483                Ok(_) => print_success(&format!("Environment '{}' deleted", name)),
484                Err(e) => print_error(&format!("Failed to save changes: {}", e)),
485            }
486        }
487
488        EnvAction::Use { name } => {
489            if env_set.get(&name).is_some() {
490                env_set.set_active(&name);
491                match storage.save_environment_set(&env_set) {
492                    Ok(_) => print_success(&format!("Active environment set to '{}'", name)),
493                    Err(e) => print_error(&format!("Failed to save changes: {}", e)),
494                }
495            } else {
496                print_error(&format!("Environment '{}' not found", name));
497            }
498        }
499    }
500}

502pub fn handle_history(action: HistoryAction) {
503    let storage = match Storage::new() {
504        Ok(s) => s,
505        Err(e) => {
506            print_error(&format!("Failed to initialize storage: {}", e));
507            return;
508        }
509    };
510
511    match action {
512        HistoryAction::List { limit } => {
513            let history = match storage.load_history() {
514                Ok(h) => h,
515                Err(e) => {
516                    print_error(&format!("Failed to load history: {}", e));
517                    return;
518                }
519            };
520
521            if history.is_empty() {
522                print_info("No history entries found");
523                return;
524            }
525
526            println!("\n{}:", "Request History".bold().cyan());
527            for (i, entry) in history.iter().take(limit).enumerate() {
528                let status_color = if entry.response.status < 300 {
529                    entry.response.status.to_string().green()
530                } else if entry.response.status < 400 {
531                    entry.response.status.to_string().yellow()
532                } else {
533                    entry.response.status.to_string().red()
534                };
535
536                println!(
537                    "\n  {} {} {} {} {}",
538                    format!("[{}]", i + 1).bright_black(),
539                    entry.request.method.bright_white(),
540                    entry.request.url,
541                    status_color,
542                    format!("({}ms)", entry.duration_ms).bright_black()
543                );
544                println!("     {}", entry.format_timestamp().bright_black());
545            }
546            println!();
547        }
548
549        HistoryAction::Show { id } => {
550            let history = match storage.load_history() {
551                Ok(h) => h,
552                Err(e) => {
553                    print_error(&format!("Failed to load history: {}", e));
554                    return;
555                }
556            };
557
558            let entry = if let Ok(index) = id.parse::<usize>() {
559                if index > 0 && index <= history.len() {
560                    &history[index - 1]
561                } else {
562                    print_error(&format!("Invalid index: {}", index));
563                    return;
564                }
565            } else {
566                match history.iter().find(|e| e.id == id) {
567                    Some(e) => e,
568                    None => {
569                        print_error(&format!("History entry not found: {}", id));
570                        return;
571                    }
572                }
573            };
574
575            println!(
576                "\n{}: {}",
577                "Request".bold().cyan(),
578                entry.format_timestamp()
579            );
580            println!(
581                "  {} {}",
582                entry.request.method.bright_white(),
583                entry.request.url
584            );
585
586            if !entry.request.headers.is_empty() {
587                println!("\n{}:", "Headers".bold());
588                for (name, value) in &entry.request.headers {
589                    println!("  {}: {}", name.bright_black(), value);
590                }
591            }
592
593            if let Some(body) = &entry.request.body {
594                println!("\n{}:", "Body".bold());
595                println!("{}", body);
596            }
597
598            println!(
599                "\n{}: {} {}",
600                "Response".bold().cyan(),
601                entry.response.status,
602                entry.response.status_text
603            );
604            println!("  Duration: {}ms", entry.duration_ms);
605
606            println!("\n{}:", "Preview".bold());
607            println!("{}", entry.response.body_preview);
608            println!();
609        }
610
611        HistoryAction::Rerun { id, verbose } => {
612            let history = match storage.load_history() {
613                Ok(h) => h,
614                Err(e) => {
615                    print_error(&format!("Failed to load history: {}", e));
616                    return;
617                }
618            };
619
620            let entry = if let Ok(index) = id.parse::<usize>() {
621                if index > 0 && index <= history.len() {
622                    &history[index - 1]
623                } else {
624                    print_error(&format!("Invalid index: {}", index));
625                    return;
626                }
627            } else {
628                match history.iter().find(|e| e.id == id) {
629                    Some(e) => e,
630                    None => {
631                        print_error(&format!("History entry not found: {}", id));
632                        return;
633                    }
634                }
635            };
636
637            println!(
638                "Rerunning: {} {}\n",
639                entry.request.method, entry.request.url
640            );
641
642            let method = match entry.request.method.to_uppercase().as_str() {
643                "GET" => HttpMethod::Get,
644                "POST" => HttpMethod::Post,
645                "PUT" => HttpMethod::Put,
646                "PATCH" => HttpMethod::Patch,
647                "DELETE" => HttpMethod::Delete,
648                "HEAD" => HttpMethod::Head,
649                "OPTIONS" => HttpMethod::Options,
650                _ => {
651                    print_error("Invalid HTTP method in history");
652                    return;
653                }
654            };
655
656            let mut request = HttpRequest::new(method, &entry.request.url);
657
658            for (key, value) in &entry.request.headers {
659                request = request.with_header(key, value);
660            }
661
662            if let Some(body) = &entry.request.body {
663                request = request.with_text_body(body);
664            }
665
666            let client = HttpClient::new();
667            let start = Instant::now();
668
669            match client.send(&request) {
670                Ok(response) => {
671                    let duration_ms = start.elapsed().as_millis() as u64;
672
673                    let new_history_request = HistoryRequest {
674                        method: entry.request.method.clone(),
675                        url: entry.request.url.clone(),
676                        headers: entry.request.headers.clone(),
677                        body: entry.request.body.clone(),
678                    };
679
680                    let history_response = HistoryResponse::from_response(&response);
681                    let new_entry =
682                        HistoryEntry::new(new_history_request, history_response, duration_ms);
683
684                    if let Err(e) = storage.save_history_entry(&new_entry) {
685                        eprintln!("Warning: Failed to save to history: {}", e);
686                    }
687
688                    print_response(&response, verbose);
689                }
690                Err(e) => print_error(&format!("Request failed: {}", e)),
691            }
692        }
693
694        HistoryAction::Clear => match storage.clear_history() {
695            Ok(_) => print_success("History cleared"),
696            Err(e) => print_error(&format!("Failed to clear history: {}", e)),
697        },
698    }
699}

701/// Converts an imported collection to native Collection format
702fn convert_imported_to_collection(imported: ImportedCollection) -> Collection {
703    let mut collection = Collection::new(imported.name);
704    if let Some(desc) = imported.description {
705        collection = collection.with_description(desc);
706    }
707
708    for imported_req in imported.requests {
709        let saved_request = SavedRequest {
710            id: imported_req.id,
711            name: imported_req.name,
712            description: imported_req.description,
713            request: SerializableRequest {
714                method: imported_req.method,
715                url: imported_req.url,
716                headers: imported_req.headers,
717                body: imported_req.body,
718                timeout_seconds: None,
719            },
720        };
721        collection.add_request(saved_request);
722    }
723
724    collection
725}
