88    pub fn send(&self, request: &HttpRequest) -> Result<HttpResponse, HttpError> {
89        let start = Instant::now();
90
91        let method_str = request.method.as_str();
92        let url = &request.url;
93
94        let mut req = self.agent.request(method_str, url);
95
96        for header in &request.headers {
97            req = req.set(&header.name, &header.value);
98        }
99
100        if let Some(timeout) = request.timeout_seconds {
101            req = req.timeout(Duration::from_secs(timeout));
102        }
103
104        let response = match &request.body {
105            RequestBody::None => req.call(),
106            RequestBody::Text(text) => {
107                if request.get_header("Content-Type").is_none() {
108                    req = req.set("Content-Type", "text/plain; charset=utf-8");
109                }
110                req.send_string(text)
111            }
112            RequestBody::Json(json) => {
113                if request.get_header("Content-Type").is_none() {
114                    req = req.set("Content-Type", "application/json; charset=utf-8");
115                }
116                req.send_string(json)
117            }
118            RequestBody::FormUrlEncoded(data) => {
119                if request.get_header("Content-Type").is_none() {
120                    req = req.set("Content-Type", "application/x-www-form-urlencoded");
121                }
122                let encoded = saffron_core::domain::request_body::encode_form_urlencoded(data);
123                req.send_string(&encoded)
124            }
125            RequestBody::Binary(bytes) => {
126                if request.get_header("Content-Type").is_none() {
127                    req = req.set("Content-Type", "application/octet-stream");
128                }
129                req.send_bytes(bytes)
130            }
131            RequestBody::FormData(parts) => {
132                return self.send_multipart(req, parts, start);
133            }
134        };
135
136        self.process_response(response, start)
137    }
138
139    fn send_multipart(
140        &self,
141        mut req: ureq::Request,
142        parts: &[FormDataPart],
143        start: Instant,
144    ) -> Result<HttpResponse, HttpError> {
145        let boundary = format!(
146            "----SaffronBoundary{}",
147            chrono::Utc::now().timestamp_millis()
148        );
149        req = req.set(
150            "Content-Type",
151            &format!("multipart/form-data; boundary={}", boundary),
152        );
153
154        let mut body = Vec::new();
155
156        for part in parts {
157            body.extend_from_slice(b"--");
158            body.extend_from_slice(boundary.as_bytes());
159            body.extend_from_slice(b"\r\n");
160
161            match &part.content {
162                FormDataContent::Text(text) => {
163                    body.extend_from_slice(
164                        format!(
165                            "Content-Disposition: form-data; name=\"{}\"\r\n\r\n",
166                            part.name
167                        )
168                        .as_bytes(),
169                    );
170                    body.extend_from_slice(text.as_bytes());
171                }
172                FormDataContent::File {
173                    filename,
174                    data,
175                    content_type,
176                } => {
177                    body.extend_from_slice(
178                        format!(
179                            "Content-Disposition: form-data; name=\"{}\"; filename=\"{}\"\r\n",
180                            part.name, filename
181                        )
182                        .as_bytes(),
183                    );
184
185                    if let Some(ct) = content_type {
186                        body.extend_from_slice(format!("Content-Type: {}\r\n", ct).as_bytes());
187                    }
188
189                    body.extend_from_slice(b"\r\n");
190                    body.extend_from_slice(data);
191                }
192            }
193            body.extend_from_slice(b"\r\n");
194        }
195
196        body.extend_from_slice(b"--");
197        body.extend_from_slice(boundary.as_bytes());
198        body.extend_from_slice(b"--\r\n");
199
200        let response = req.send_bytes(&body);
201        self.process_response(response, start)
202    }

204    fn process_response(
205        &self,
206        response: Result<ureq::Response, ureq::Error>,
207        start: Instant,
208    ) -> Result<HttpResponse, HttpError> {
209        let elapsed = start.elapsed();
210
211        match response {
212            Ok(resp) => self.extract_response(resp, elapsed),
213            Err(ureq::Error::Status(code, resp)) => {
214                self.extract_response_with_code(resp, code, elapsed)
215            }
216            Err(ureq::Error::Transport(transport)) => {
217                let error_msg = transport.to_string();
218                if error_msg.contains("timeout") || error_msg.contains("timed out") {
219                    Err(HttpError::Timeout)
220                } else if error_msg.contains("certificate") || error_msg.contains("tls") {
221                    Err(HttpError::TlsError(error_msg))
222                } else {
223                    Err(HttpError::NetworkError(error_msg))
224                }
225            }
226        }
227    }

229    fn extract_response(
230        &self,
231        resp: ureq::Response,
232        elapsed: Duration,
233    ) -> Result<HttpResponse, HttpError> {
234        let status = resp.status();
235        let status_text = resp.status_text().to_string();
236        let url = resp.get_url().to_string();
237
238        let mut headers = HashMap::new();
239        for name in resp.headers_names() {
240            if let Some(value) = resp.header(&name) {
241                headers.insert(name.clone(), value.to_string());
242            }
243        }
244
245        let body = self.read_response_body(resp)?;
246
247        Ok(HttpResponse::new(
248            status,
249            status_text,
250            headers,
251            body,
252            elapsed,
253            url,
254        ))
255    }

285    fn read_response_body(&self, resp: ureq::Response) -> Result<Vec<u8>, HttpError> {
286        let mut reader = resp.into_reader();
287        let mut body = Vec::new();
288
289        if let Some(max_size) = self.config.max_response_size {
290            let mut limited_reader = reader.take(max_size as u64);
291            limited_reader.read_to_end(&mut body)?;
292        } else {
293            reader.read_to_end(&mut body)?;
294        }
295
296        Ok(body)
297    }

343    pub fn download_file(url: &str, path: &std::path::Path) -> Result<(), HttpError> {
344        let client = HttpClient::new();

352    pub fn upload_file(
353        url: &str,
354        field_name: &str,
355        file_path: &std::path::Path,
356    ) -> Result<HttpResponse, HttpError> {
357        let filename = file_path
358            .file_name()
359            .and_then(|n| n.to_str())
360            .unwrap_or("file")
361            .to_string();
362
363        let data = std::fs::read(file_path)?;
364
365        let part = FormDataPart {
366            name: field_name.to_string(),
367            content: FormDataContent::File {
368                filename,
369                data,
370                content_type: Some(guess_content_type(file_path)),
371            },
372        };
373
374        let request = HttpRequest::post(url).with_body(RequestBody::FormData(vec![part]));
375
376        let client = HttpClient::new();
377        client.send(&request)
378    }

380    fn guess_content_type(path: &std::path::Path) -> String {
381        match path.extension().and_then(|e| e.to_str()) {
382            Some("txt") => "text/plain",
383            Some("html") | Some("htm") => "text/html",
384            Some("json") => "application/json",
385            Some("xml") => "application/xml",
386            Some("pdf") => "application/pdf",
387            Some("jpg") | Some("jpeg") => "image/jpeg",
388            Some("png") => "image/png",
389            Some("gif") => "image/gif",
390            Some("svg") => "image/svg+xml",
391            Some("mp4") => "video/mp4",
392            Some("zip") => "application/zip",
393            _ => "application/octet-stream",
394        }
395        .to_string()
396    }
